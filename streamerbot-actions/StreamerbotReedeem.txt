using System;
using System.Collections.Generic;
using System.IO;
using System.Text;
using System.Threading;
using Newtonsoft.Json;
using Newtonsoft.Json.Linq;

public class CPHInline
{
    private const string RedeemSuccessTemplate = "⚡ Scan complete: @{viewer} found {item} [{collection}]. Progress: {owned}/{total}.{duplicateText}";
    private const string RarePullTemplate = "{rareLabel}: @{viewer} pulled {item}! Current odds: about 1 in {odds}.";
    private const string TriplePullTemplate = "TRIPLE MATCH: @{viewer} pulled {item} three times in a row! Sequence odds: about 1 in {odds}.";
    private const string CollectionCompleteTemplate = "✅ COLLECTION COMPLETE: @{viewer} completed {collection}!";
    private const string VariantPullTemplate = "";

    public bool Execute()
    {
        string folderPath = @"C:\CircuitOS\Data";

        string componentsPath = Path.Combine(folderPath, "components.json");
        string featuredBoostPath = Path.Combine(folderPath, "featured-boost.json");
        string inventoryPath = Path.Combine(folderPath, "inventory.json");
        string inventoryLockPath = Path.Combine(folderPath, "inventory.lock");
        string backupFolderPath = Path.Combine(folderPath, "backups");
        string overlayFolderPath = Path.Combine(folderPath, "overlay");
        string overlayStatePath = Path.Combine(overlayFolderPath, "overlay-state.json");

        CPH.LogInfo("[Circuit Component] Action started.");
        CPH.LogInfo("[Circuit Component] Folder path: " + folderPath);

        if (!Directory.Exists(folderPath))
        {
            Directory.CreateDirectory(folderPath);
            CPH.LogInfo("[Circuit Component] Folder did not exist, so it was created.");
        }

        if (!File.Exists(componentsPath))
        {
            CPH.SendMessage("⚠️ Circuit Component error: components.json was not found.", true, true);
            CPH.LogError("[Circuit Component] components.json missing at: " + componentsPath);
            return false;
        }

        JObject catalog;
        JObject featuredBoost;
        JObject inventory;

        try
        {
            catalog = JObject.Parse(File.ReadAllText(componentsPath));
        }
        catch (Exception ex)
        {
            CPH.SendMessage("⚠️ Circuit Component error: components.json is not valid JSON.", true, true);
            CPH.LogError("[Circuit Component] components.json parse error: " + ex.Message);
            return false;
        }

        try
        {
            featuredBoost = LoadFeaturedBoost(featuredBoostPath);
        }
        catch (Exception ex)
        {
            CPH.SendMessage("Circuit Component error: featured-boost.json is invalid.", true, true);
            CPH.LogError("[Circuit Component] Featured boost config error: " + ex.Message);
            return false;
        }

        string viewerId = GetArgOrDefault("userId", "test_user_001");
        string viewerName = GetArgOrDefault("userName", "");
        if (string.IsNullOrWhiteSpace(viewerName)) viewerName = GetArgOrDefault("displayName", "");
        if (string.IsNullOrWhiteSpace(viewerName)) viewerName = GetArgOrDefault("user", "TestViewer");

        // Per-user soft cooldown — refunds points if Streamer.bot supports it
        const int CooldownSeconds = 120;
        // After a duplicate pull, guarantee this many non-duplicate pulls (0 = disabled)
        const int DupProtectionTurns = 0;
        string cooldownKey = "cc_cd_" + viewerId;
        string lastPullUtc = CPH.GetGlobalVar<string>(cooldownKey, false);
        if (!string.IsNullOrEmpty(lastPullUtc) &&
            DateTime.TryParse(lastPullUtc, null, System.Globalization.DateTimeStyles.RoundtripKind, out DateTime lastPullTime))
        {
            int elapsed = (int)(DateTime.UtcNow - lastPullTime).TotalSeconds;
            if (elapsed < CooldownSeconds)
            {
                int wait = CooldownSeconds - elapsed;
                CPH.SendMessage($"@{viewerName} Scan on cooldown — try again in {wait} more second{(wait == 1 ? "" : "s")}.", true, true);
                CPH.LogInfo($"[Circuit Component] Cooldown: {viewerName} ({viewerId}) must wait {wait}s.");
               
                return false;
            }
        }

        FileStream inventoryLock = AcquireInventoryLock(inventoryLockPath);

        if (inventoryLock == null)
        {
            CPH.SendMessage("Circuit Component error: inventory is busy. Please try again.", true, true);
            CPH.LogError("[Circuit Component] Could not acquire the inventory lock.");
            return false;
        }

        using (inventoryLock)
        {
        try
        {
            inventory = LoadInventory(inventoryPath);
        }
        catch (Exception ex)
        {
            CPH.SendMessage("⚠️ Circuit Component error: inventory.json is not valid JSON.", true, true);
            CPH.LogError("[Circuit Component] inventory.json parse error: " + ex.Message);
            return false;
        }

        JObject collections = catalog["collections"] as JObject;

        if (collections == null)
        {
            CPH.SendMessage("⚠️ Circuit Component error: components.json needs a top-level 'collections' object.", true, true);
            CPH.LogError("[Circuit Component] Missing collections object.");
            return false;
        }

        string activeBoostName;
        string collectionKey;
        double selectedCollectionProbability;

        try
        {
            collectionKey = RollCollection(
                collections,
                featuredBoost,
                out activeBoostName,
                out selectedCollectionProbability
            );
        }
        catch (Exception ex)
        {
            CPH.SendMessage("Circuit Component error: collection weights are invalid.", true, true);
            CPH.LogError("[Circuit Component] Collection roll error: " + ex.Message);
            return false;
        }

        if (!string.IsNullOrWhiteSpace(activeBoostName))
        {
            JObject boostMultipliers = featuredBoost?["collectionMultipliers"] as JObject;
            if (boostMultipliers?[collectionKey] == null)
                activeBoostName = "";
            else
                CPH.LogInfo("[Circuit Component] Featured boost active: " + activeBoostName);
        }

        JObject selectedCollection = collections[collectionKey] as JObject;

        if (selectedCollection == null)
        {
            CPH.SendMessage($"⚠️ Circuit Component error: collection '{collectionKey}' does not exist in components.json.", true, true);
            CPH.LogError("[Circuit Component] Missing collection: " + collectionKey);
            return false;
        }

        string collectionName = selectedCollection["displayName"]?.ToString() ?? collectionKey;
        JArray parts = selectedCollection["parts"] as JArray;

        if (parts == null || parts.Count == 0)
        {
            CPH.SendMessage($"⚠️ Circuit Component error: {collectionName} has no parts.", true, true);
            CPH.LogError("[Circuit Component] No parts found for: " + collectionKey);
            return false;
        }

        // Read dup protection state before rolling (viewer entry may not exist yet)
        int pullsSinceLastDup = DupProtectionTurns; // default = protection not active
        if (DupProtectionTurns > 0 && inventory[viewerId] != null)
        {
            JToken psld = inventory[viewerId]["pullsSinceLastDup"];
            if (psld != null) Int32.TryParse(psld.ToString(), out pullsSinceLastDup);
        }

        // If protection is active, restrict rolls to unowned parts only
        JArray eligibleParts = parts;
        if (DupProtectionTurns > 0 && pullsSinceLastDup < DupProtectionTurns)
        {
            JObject preCheck = (inventory[viewerId]?["components"] as JObject) ?? new JObject();
            JArray unowned = new JArray();
            foreach (JToken pt in parts)
            {
                JObject p = pt as JObject;
                string pid = p?["id"]?.ToString();
                if (!string.IsNullOrEmpty(pid) && (preCheck[pid] == null || preCheck[pid].Value<int>() == 0))
                    unowned.Add(p);
            }
            if (unowned.Count > 0) eligibleParts = unowned;
        }

        Random random = new Random(Guid.NewGuid().GetHashCode());
        JObject selectedPart;
        double partPullProbability;
        string selectedTierLabel = "";

        JArray collectionTiers = selectedCollection["tiers"] as JArray;
        if (collectionTiers != null && collectionTiers.Count > 0)
        {
            // Group eligible parts by tier, skipping tiers whose items are all owned
            var tierGroups = new Dictionary<string, List<JObject>>(StringComparer.Ordinal);
            var tierWeightMap = new Dictionary<string, double>(StringComparer.Ordinal);
            double totalTierWeight = 0;
            foreach (JObject tier in collectionTiers)
            {
                string tid = tier["id"]?.ToString() ?? "";
                double tw = 0;
                Double.TryParse(tier["weight"]?.ToString(),
                    System.Globalization.NumberStyles.Float,
                    System.Globalization.CultureInfo.InvariantCulture, out tw);
                if (string.IsNullOrEmpty(tid) || tw <= 0) continue;
                var group = new List<JObject>();
                foreach (JToken pt in eligibleParts)
                {
                    JObject p = pt as JObject;
                    if (p?["tier"]?.ToString() == tid) group.Add(p);
                }
                if (group.Count == 0) continue;
                tierGroups[tid] = group;
                tierWeightMap[tid] = tw;
                totalTierWeight += tw;
            }

            if (tierGroups.Count == 0 || totalTierWeight <= 0)
            {
                // Fallback: equal odds across all eligible parts
                selectedPart = eligibleParts[random.Next(eligibleParts.Count)] as JObject;
                partPullProbability = selectedCollectionProbability / eligibleParts.Count;
            }
            else
            {
                // Roll tier
                double tierRoll = random.NextDouble() * totalTierWeight;
                double cumulative = 0;
                string chosenTierId = "";
                double chosenTierWeight = 0;
                foreach (var kv in tierWeightMap)
                {
                    cumulative += kv.Value;
                    if (tierRoll < cumulative || chosenTierId == "")
                    {
                        chosenTierId = kv.Key;
                        chosenTierWeight = kv.Value;
                        if (tierRoll < cumulative) break;
                    }
                }
                // Find tier label for overlay/chat
                foreach (JObject tier in collectionTiers)
                {
                    if (tier["id"]?.ToString() == chosenTierId)
                    {
                        selectedTierLabel = tier["label"]?.ToString() ?? "";
                        break;
                    }
                }
                // Roll item within tier
                var chosenGroup = tierGroups[chosenTierId];
                selectedPart = chosenGroup[random.Next(chosenGroup.Count)];
                partPullProbability = selectedCollectionProbability
                    * (chosenTierWeight / totalTierWeight)
                    / chosenGroup.Count;
            }
        }
        else
        {
            // No tiers: equal odds across all eligible parts
            selectedPart = eligibleParts[random.Next(eligibleParts.Count)] as JObject;
            partPullProbability = selectedCollectionProbability / eligibleParts.Count;
        }

        string partId = selectedPart["id"]?.ToString();
        string partName = selectedPart["name"]?.ToString();

        if (string.IsNullOrWhiteSpace(partId) || string.IsNullOrWhiteSpace(partName))
        {
            CPH.SendMessage("⚠️ Circuit Component error: selected part is missing an id or name.", true, true);
            CPH.LogError("[Circuit Component] Bad part data.");
            return false;
        }

        // Roll variants — independent sequential rolls, cap at 2, no duplicate labels
        var variantLabels = new List<string>();
        JArray collectionVariants = selectedCollection["variants"] as JArray;
        if (collectionVariants != null)
        {
            Random variantRng = new Random(Guid.NewGuid().GetHashCode());
            foreach (JToken vt in collectionVariants)
            {
                if (variantLabels.Count >= 2) break;
                JObject variant = vt as JObject;
                string vlabel = variant?["label"]?.ToString() ?? "";
                double vchance = 0;
                Double.TryParse(variant?["chance"]?.ToString(),
                    System.Globalization.NumberStyles.Float,
                    System.Globalization.CultureInfo.InvariantCulture, out vchance);
                if (string.IsNullOrWhiteSpace(vlabel) || vchance <= 0) continue;
                if (!variantLabels.Contains(vlabel) && variantRng.NextDouble() < vchance)
                    variantLabels.Add(vlabel);
            }
        }
        string variantPrefix = variantLabels.Count > 0 ? string.Join(" ", variantLabels) + " " : "";
        string displayPartName = variantPrefix + partName;

        if (inventory[viewerId] == null)
        {
            inventory[viewerId] = new JObject
            {
                ["displayName"] = viewerName,
                ["components"] = new JObject()
            };
        }

        inventory[viewerId]["displayName"] = viewerName;

        JObject viewerComponents = inventory[viewerId]["components"] as JObject;

        if (viewerComponents == null)
        {
            viewerComponents = new JObject();
            inventory[viewerId]["components"] = viewerComponents;
        }

        JToken completedCollectionsToken = inventory[viewerId]["completedCollections"];
        JObject completedCollections;

        if (completedCollectionsToken == null)
        {
            completedCollections = new JObject();
            inventory[viewerId]["completedCollections"] = completedCollections;
        }
        else
        {
            completedCollections = completedCollectionsToken as JObject;

            if (completedCollections == null)
            {
                CPH.SendMessage("Circuit Component error: completion data is invalid.", true, true);
                CPH.LogError("[Circuit Component] completedCollections must be a JSON object for viewer " + viewerId + ".");
                return false;
            }
        }

        int seededCompletions = SeedExistingCompletions(
            viewerComponents,
            collections,
            completedCollections
        );

        if (seededCompletions > 0)
        {
            CPH.LogInfo(
                "[Circuit Component] Silently recorded " + seededCompletions +
                " existing completion(s) for " + viewerName + "."
            );
        }

        int ownedBefore = CountOwnedParts(viewerComponents, parts);

        int quantity = 0;

        if (viewerComponents[partId] != null)
        {
            quantity = viewerComponents[partId].Value<int>();
        }

        quantity++;
        viewerComponents[partId] = quantity;

        // Update dup protection counter (quantity > 1 means this pull was a duplicate)
        if (DupProtectionTurns > 0)
        {
            JObject viewerObj2 = inventory[viewerId] as JObject;
            if (viewerObj2 != null)
                viewerObj2["pullsSinceLastDup"] = (quantity > 1) ? 0 : Math.Min(pullsSinceLastDup + 1, DupProtectionTurns);
        }

        int consecutivePullCount;
        double streakSequenceProbability;

        UpdatePullStreak(
            inventory[viewerId] as JObject,
            partId,
            partPullProbability,
            out consecutivePullCount,
            out streakSequenceProbability
        );

        int ownedAfter = CountOwnedParts(viewerComponents, parts);
        int totalParts = parts.Count;
        bool newlyCompleted = false;

        if (ownedBefore < totalParts &&
            ownedAfter == totalParts &&
            completedCollections[collectionKey] == null)
        {
            completedCollections[collectionKey] = DateTime.UtcNow.ToString("yyyy-MM-ddTHH:mm:ss.fffZ");
            newlyCompleted = true;
        }

        string rareLabel = GetRarePullLabel(collectionKey, selectedCollection);

        try
        {
            string backupPath = SaveInventorySafely(inventoryPath, backupFolderPath, inventory);

            if (string.IsNullOrWhiteSpace(backupPath))
            {
                CPH.LogInfo("[Circuit Component] Created a new inventory.json file.");
            }
            else
            {
                CPH.LogInfo("[Circuit Component] Inventory backup created: " + backupPath);
            }
        }
        catch (Exception ex)
        {
            CPH.SendMessage("⚠️ Circuit Component error: could not save inventory.json.", true, true);
            CPH.LogError("[Circuit Component] Save error: " + ex.Message);
            return false;
        }

        try
        {
            SaveOverlayStateSafely(
                overlayFolderPath,
                overlayStatePath,
                viewerName,
                partName,
                collectionKey,
                collectionName,
                ownedAfter,
                totalParts,
                quantity,
                newlyCompleted,
                rareLabel,
                activeBoostName,
                variantLabels,
                selectedTierLabel
            );
        }
        catch (Exception ex)
        {
            CPH.LogError("[Circuit Component] Overlay update warning: " + ex.Message);
        }

        CPH.SetGlobalVar(cooldownKey, DateTime.UtcNow.ToString("O"), false);

        string duplicateText = quantity > 1 ? $" Duplicate count: x{quantity}." : "";

        if (!string.IsNullOrWhiteSpace(activeBoostName))
        {
            duplicateText += $" Featured boost: {activeBoostName}.";
        }

        CPH.SendMessage(
            FormatMessage(RedeemSuccessTemplate,
                "viewer", viewerName,
                "item", displayPartName,
                "collection", collectionName,
                "owned", ownedAfter.ToString(),
                "total", totalParts.ToString(),
                "duplicateText", duplicateText),
            true,
            true
        );

        if (!string.IsNullOrWhiteSpace(rareLabel))
        {
            CPH.SendMessage(
                FormatMessage(RarePullTemplate,
                    "rareLabel", rareLabel,
                    "viewer", viewerName,
                    "item", displayPartName,
                    "odds", FormatOneInOdds(partPullProbability)),
                true,
                true
            );
        }

        if (consecutivePullCount == 3)
        {
            CPH.SendMessage(
                FormatMessage(TriplePullTemplate,
                    "viewer", viewerName,
                    "item", displayPartName,
                    "odds", FormatOneInOdds(streakSequenceProbability)),
                true,
                true
            );
        }

        if (newlyCompleted)
        {
            CPH.SendMessage(
                FormatMessage(CollectionCompleteTemplate,
                    "viewer", viewerName,
                    "collection", collectionName),
                true,
                true
            );
        }

        if (variantLabels.Count > 0 && !string.IsNullOrWhiteSpace(VariantPullTemplate))
        {
            // {item} here is the BASE name so a template can compose
            // "{variantLabels} {item}" without doubling the variant labels.
            CPH.SendMessage(
                FormatMessage(VariantPullTemplate,
                    "variantLabels", string.Join(" ", variantLabels),
                    "viewer", viewerName,
                    "item", partName,
                    "collection", collectionName),
                true,
                true
            );
        }

        CPH.LogInfo($"[Circuit Component] {viewerName} found {displayPartName} [{collectionName}]. Progress {ownedAfter}/{totalParts}.");

        return true;
    }
    }

    private string FormatMessage(string template, params string[] replacements)
    {
        string result = template ?? "";
        for (int i = 0; i + 1 < replacements.Length; i += 2)
        {
            result = result.Replace("{" + replacements[i] + "}", replacements[i + 1] ?? "");
        }
        return result;
    }

    private string GetArgOrDefault(string argName, string defaultValue)
    {
        string value;

        if (CPH.TryGetArg(argName, out value))
        {
            if (!string.IsNullOrWhiteSpace(value))
            {
                return value;
            }
        }

        return defaultValue;
    }

    private int CountOwnedParts(JObject viewerComponents, JArray parts)
    {
        int count = 0;

        foreach (JObject part in parts)
        {
            string id = part["id"]?.ToString();

            if (!string.IsNullOrWhiteSpace(id) &&
                viewerComponents[id] != null &&
                viewerComponents[id].Value<int>() > 0)
            {
                count++;
            }
        }

        return count;
    }

    private void UpdatePullStreak(
        JObject viewer,
        string partId,
        double partPullProbability,
        out int consecutivePullCount,
        out double sequenceProbability)
    {
        JObject pullStreak = viewer["pullStreak"] as JObject;

        if (pullStreak == null)
        {
            pullStreak = new JObject();
            viewer["pullStreak"] = pullStreak;
        }

        string previousPartId = pullStreak["partId"]?.ToString() ?? "";
        int previousCount = 0;
        double previousSequenceProbability = 0;

        if (pullStreak["count"] != null)
        {
            Int32.TryParse(pullStreak["count"].ToString(), out previousCount);
        }

        if (pullStreak["sequenceProbability"] != null)
        {
            Double.TryParse(
                pullStreak["sequenceProbability"].ToString(),
                System.Globalization.NumberStyles.Float,
                System.Globalization.CultureInfo.InvariantCulture,
                out previousSequenceProbability
            );
        }

        if (previousPartId == partId &&
            previousCount > 0 &&
            previousSequenceProbability > 0)
        {
            consecutivePullCount = previousCount + 1;
            sequenceProbability = previousSequenceProbability * partPullProbability;
        }
        else
        {
            consecutivePullCount = 1;
            sequenceProbability = partPullProbability;
        }

        pullStreak["partId"] = partId;
        pullStreak["count"] = consecutivePullCount;
        pullStreak["sequenceProbability"] = sequenceProbability;
    }

    private string GetRarePullLabel(string collectionKey, JObject collection)
    {
        string configuredLabel = collection["rareLabel"]?.ToString() ?? "";

        if (!string.IsNullOrWhiteSpace(configuredLabel))
        {
            return configuredLabel;
        }

        // Preserve rare messages while older live catalogs are upgraded.
        if (collectionKey == "broken") return "RARE CIRCUIT PULL";
        if (collectionKey == "quantum") return "QUANTUM PULL";

        return "";
    }

    private string FormatOneInOdds(double probability)
    {
        if (probability <= 0)
        {
            return "unknown";
        }

        double oneIn = Math.Max(1, Math.Round(1.0 / probability));
        return oneIn.ToString("N0");
    }

    private int SeedExistingCompletions(
        JObject viewerComponents,
        JObject collections,
        JObject completedCollections)
    {
        int seededCount = 0;
        string recordedAtUtc = DateTime.UtcNow.ToString("yyyy-MM-ddTHH:mm:ss.fffZ");

        foreach (JProperty collectionProperty in collections.Properties())
        {
            if (completedCollections[collectionProperty.Name] != null)
            {
                continue;
            }

            JObject collection = collectionProperty.Value as JObject;
            JArray collectionParts = collection?["parts"] as JArray;

            if (collectionParts == null || collectionParts.Count == 0)
            {
                continue;
            }

            if (CountOwnedParts(viewerComponents, collectionParts) == collectionParts.Count)
            {
                completedCollections[collectionProperty.Name] = recordedAtUtc;
                seededCount++;
            }
        }

        return seededCount;
    }

    private FileStream AcquireInventoryLock(string lockPath)
    {
        const int attempts = 50;
        const int delayMilliseconds = 100;

        for (int attempt = 1; attempt <= attempts; attempt++)
        {
            try
            {
                return new FileStream(
                    lockPath,
                    FileMode.OpenOrCreate,
                    FileAccess.ReadWrite,
                    FileShare.None
                );
            }
            catch (IOException)
            {
                if (attempt < attempts)
                {
                    Thread.Sleep(delayMilliseconds);
                }
            }
        }

        return null;
    }

    private JObject LoadInventory(string inventoryPath)
    {
        if (!File.Exists(inventoryPath))
        {
            return new JObject();
        }

        string inventoryText = File.ReadAllText(inventoryPath);

        if (string.IsNullOrWhiteSpace(inventoryText))
        {
            throw new Exception("inventory.json is empty.");
        }

        return JObject.Parse(inventoryText);
    }

    private string SaveInventorySafely(string inventoryPath, string backupFolderPath, JObject inventory)
    {
        string inventoryFolderPath = Path.GetDirectoryName(inventoryPath);
        string tempPath = Path.Combine(
            inventoryFolderPath,
            "inventory." + Guid.NewGuid().ToString("N") + ".tmp"
        );

        try
        {
            string inventoryJson = inventory.ToString(Formatting.Indented);

            // Validate both the serialized text and the bytes written to disk.
            JObject.Parse(inventoryJson);
            File.WriteAllText(tempPath, inventoryJson, new UTF8Encoding(false));
            JObject.Parse(File.ReadAllText(tempPath));

            if (!File.Exists(inventoryPath))
            {
                File.Move(tempPath, inventoryPath);
                return "";
            }

            Directory.CreateDirectory(backupFolderPath);

            string backupPath = Path.Combine(
                backupFolderPath,
                "inventory_" + DateTime.Now.ToString("yyyyMMdd_HHmmss_fff") + ".json"
            );

            File.Replace(tempPath, inventoryPath, backupPath);
            return backupPath;
        }
        finally
        {
            if (File.Exists(tempPath))
            {
                try
                {
                    File.Delete(tempPath);
                }
                catch
                {
                    // A leftover temp file does not affect the saved inventory.
                }
            }
        }
    }

    private void SaveOverlayStateSafely(
        string overlayFolderPath,
        string overlayStatePath,
        string viewerName,
        string partName,
        string collectionKey,
        string collectionName,
        int ownedCount,
        int totalCount,
        int quantity,
        bool newlyCompleted,
        string rareLabel,
        string activeBoostName,
        List<string> variantLabels,
        string tierLabel)
    {
        Directory.CreateDirectory(overlayFolderPath);

        JArray variantLabelsJson = new JArray();
        foreach (string v in (variantLabels ?? new List<string>())) variantLabelsJson.Add(v);

        JObject state = new JObject
        {
            ["version"] = 1,
            ["updatedAtUtc"] = DateTime.UtcNow.ToString("yyyy-MM-ddTHH:mm:ss.fffZ"),
            ["visibleUntilUtc"] = DateTime.UtcNow.AddSeconds(20).ToString("yyyy-MM-ddTHH:mm:ss.fffZ"),
            ["viewerName"] = viewerName,
            ["partName"] = partName,
            ["collectionKey"] = collectionKey,
            ["collectionName"] = collectionName,
            ["ownedCount"] = ownedCount,
            ["totalCount"] = totalCount,
            ["quantity"] = quantity,
            ["isDuplicate"] = quantity > 1,
            ["newlyCompleted"] = newlyCompleted,
            ["rareLabel"] = rareLabel ?? "",
            ["featuredBoost"] = activeBoostName ?? "",
            ["variantLabels"] = variantLabelsJson,
            ["tierLabel"] = tierLabel ?? ""
        };

        string tempPath = Path.Combine(
            overlayFolderPath,
            "overlay-state." + Guid.NewGuid().ToString("N") + ".tmp"
        );
        string previousStatePath = Path.Combine(
            overlayFolderPath,
            "overlay-state.previous." + Guid.NewGuid().ToString("N") + ".tmp"
        );

        try
        {
            string stateJson = state.ToString(Formatting.Indented);
            JObject.Parse(stateJson);
            File.WriteAllText(tempPath, stateJson, new UTF8Encoding(false));
            JObject.Parse(File.ReadAllText(tempPath));

            if (File.Exists(overlayStatePath))
            {
                File.Replace(tempPath, overlayStatePath, previousStatePath);
            }
            else
            {
                File.Move(tempPath, overlayStatePath);
            }
        }
        finally
        {
            if (File.Exists(tempPath))
            {
                try
                {
                    File.Delete(tempPath);
                }
                catch
                {
                    // A stale overlay temp file does not affect inventory data.
                }
            }

            if (File.Exists(previousStatePath))
            {
                try
                {
                    File.Delete(previousStatePath);
                }
                catch
                {
                    // The previous overlay state is disposable display data.
                }
            }
        }
    }

    private JObject LoadFeaturedBoost(string featuredBoostPath)
    {
        if (!File.Exists(featuredBoostPath))
        {
            return null;
        }

        string configText = File.ReadAllText(featuredBoostPath);

        if (string.IsNullOrWhiteSpace(configText))
        {
            throw new Exception("featured-boost.json is empty.");
        }

        return JObject.Parse(configText);
    }

    private string RollCollection(
        JObject collections,
        JObject featuredBoost,
        out string activeBoostName,
        out double selectedCollectionProbability)
    {
        activeBoostName = "";
        selectedCollectionProbability = 0;
        JObject multipliers = null;
        bool boostEnabled = featuredBoost?["enabled"]?.Value<bool>() ?? false;

        if (boostEnabled)
        {
            activeBoostName = featuredBoost["displayName"]?.ToString();
            multipliers = featuredBoost["collectionMultipliers"] as JObject;

            if (string.IsNullOrWhiteSpace(activeBoostName))
            {
                throw new Exception("An enabled boost needs a displayName.");
            }

            if (multipliers == null || !multipliers.HasValues)
            {
                throw new Exception("An enabled boost needs collectionMultipliers.");
            }

            foreach (JProperty multiplierProperty in multipliers.Properties())
            {
                if (collections[multiplierProperty.Name] == null)
                {
                    throw new Exception(
                        "Boost references unknown collection: " + multiplierProperty.Name
                    );
                }

                double multiplier = multiplierProperty.Value.Value<double>();

                if (multiplier <= 0)
                {
                    throw new Exception("Boost multipliers must be greater than zero.");
                }
            }
        }

        List<string> collectionKeys = new List<string>();
        List<double> effectiveWeights = new List<double>();
        double totalWeight = 0;

        foreach (JProperty collectionProperty in collections.Properties())
        {
            JObject collection = collectionProperty.Value as JObject;

            if (collection == null)
            {
                continue;
            }

            if (!IsCollectionActive(collectionProperty.Name, collection, DateTimeOffset.UtcNow))
            {
                continue;
            }

            double weight = GetCollectionWeight(collectionProperty.Name, collection);

            if (multipliers?[collectionProperty.Name] != null)
            {
                weight *= multipliers[collectionProperty.Name].Value<double>();
            }

            if (weight <= 0)
            {
                continue;
            }

            collectionKeys.Add(collectionProperty.Name);
            effectiveWeights.Add(weight);
            totalWeight += weight;
        }

        if (collectionKeys.Count == 0 || totalWeight <= 0)
        {
            throw new Exception("No collections have a positive weight.");
        }

        Random random = new Random(Guid.NewGuid().GetHashCode());
        double roll = random.NextDouble() * totalWeight;
        double cumulativeWeight = 0;

        for (int i = 0; i < collectionKeys.Count; i++)
        {
            cumulativeWeight += effectiveWeights[i];

            if (roll < cumulativeWeight)
            {
                selectedCollectionProbability = effectiveWeights[i] / totalWeight;
                return collectionKeys[i];
            }
        }

        selectedCollectionProbability = effectiveWeights[effectiveWeights.Count - 1] / totalWeight;
        return collectionKeys[collectionKeys.Count - 1];
    }

    private bool IsCollectionActive(
        string collectionKey,
        JObject collection,
        DateTimeOffset utcNow)
    {
        JToken enabledToken = collection["enabled"];

        if (enabledToken != null && !enabledToken.Value<bool>())
        {
            return false;
        }

        string collectionType = collection["type"]?.ToString() ?? "permanent";

        if (collectionType.Equals("permanent", StringComparison.OrdinalIgnoreCase))
        {
            return true;
        }

        if (!collectionType.Equals("event", StringComparison.OrdinalIgnoreCase))
        {
            throw new Exception(
                "Unknown collection type for " + collectionKey + ": " + collectionType
            );
        }

        if (enabledToken == null)
        {
            return false;
        }

        string activeFromText = collection["activeFromUtc"]?.ToString() ?? "";
        string activeUntilText = collection["activeUntilUtc"]?.ToString() ?? "";
        DateTimeOffset activeFrom;
        DateTimeOffset activeUntil;

        bool validFrom = DateTimeOffset.TryParse(
            activeFromText,
            System.Globalization.CultureInfo.InvariantCulture,
            System.Globalization.DateTimeStyles.AssumeUniversal |
                System.Globalization.DateTimeStyles.AdjustToUniversal,
            out activeFrom
        );

        bool validUntil = DateTimeOffset.TryParse(
            activeUntilText,
            System.Globalization.CultureInfo.InvariantCulture,
            System.Globalization.DateTimeStyles.AssumeUniversal |
                System.Globalization.DateTimeStyles.AdjustToUniversal,
            out activeUntil
        );

        if (!validFrom || !validUntil || activeUntil <= activeFrom)
        {
            throw new Exception(
                "Event collection has an invalid UTC schedule: " + collectionKey
            );
        }

        return utcNow >= activeFrom && utcNow < activeUntil;
    }

    private double GetCollectionWeight(string collectionKey, JObject collection)
    {
        JToken configuredWeight = collection["weight"];

        if (configuredWeight != null)
        {
            double weight = configuredWeight.Value<double>();

            if (weight < 0)
            {
                throw new Exception(
                    "Collection weight cannot be negative: " + collectionKey
                );
            }

            return weight;
        }

        // Preserve the original odds while older live catalogs are upgraded.
        if (collectionKey == "basic") return 45;
        if (collectionKey == "power") return 25;
        if (collectionKey == "advanced") return 15;
        if (collectionKey == "broken") return 10;
        if (collectionKey == "quantum") return 5;

        throw new Exception("Collection is missing a weight: " + collectionKey);
    }
