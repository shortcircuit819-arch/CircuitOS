using System;
using System.IO;
using System.Text;

public class CPHInline
{
    public bool Execute()
    {
        string folderPath = @"C:\Users\nicho\OneDrive\Documents\CircuitComponents";
        string inventoryPath = Path.Combine(folderPath, "inventory.json");

        if (!File.Exists(inventoryPath))
        {
            CPH.SendMessage("⚠️ Dupes command error: inventory.json was not found.", true, true);
            return false;
        }

        string viewerId = GetArgOrDefault("userId", "");
        string viewerName = GetArgOrDefault("userName", "");

        if (string.IsNullOrWhiteSpace(viewerName))
        {
            viewerName = GetArgOrDefault("displayName", "");
        }

        if (string.IsNullOrWhiteSpace(viewerName))
        {
            viewerName = GetArgOrDefault("user", "Viewer");
        }

        if (string.IsNullOrWhiteSpace(viewerId))
        {
            CPH.SendMessage("@" + viewerName + " I couldn't find your Twitch user ID yet. Try redeeming a Circuit Component first.", true, true);
            return false;
        }

        string inventoryText = File.ReadAllText(inventoryPath);
        string viewerBlock = ExtractJsonObjectForKey(inventoryText, viewerId);

        if (string.IsNullOrWhiteSpace(viewerBlock))
        {
            CPH.SendMessage("@" + viewerName + " you don't have any Circuit Components yet. Redeem Circuit Component to start your collection.", true, true);
            return true;
        }

        string componentsBlock = ExtractJsonObjectForKey(viewerBlock, "components");

        if (string.IsNullOrWhiteSpace(componentsBlock))
        {
            CPH.SendMessage("@" + viewerName + " you don't have any Circuit Components yet. Redeem Circuit Component to start your collection.", true, true);
            return true;
        }

        StringBuilder dupes = new StringBuilder();

        AddDuplicate(dupes, componentsBlock, "basic_resistor", "Basic Resistor");
        AddDuplicate(dupes, componentsBlock, "basic_capacitor", "Basic Capacitor");
        AddDuplicate(dupes, componentsBlock, "basic_wire", "Basic Wire");
        AddDuplicate(dupes, componentsBlock, "basic_led", "Basic LED");
        AddDuplicate(dupes, componentsBlock, "basic_switch", "Basic Switch");

        AddDuplicate(dupes, componentsBlock, "power_battery", "Power Battery");
        AddDuplicate(dupes, componentsBlock, "power_capacitor", "Power Capacitor");
        AddDuplicate(dupes, componentsBlock, "power_fuse", "Power Fuse");
        AddDuplicate(dupes, componentsBlock, "power_switch", "Power Switch");
        AddDuplicate(dupes, componentsBlock, "power_surge_protector", "Power Surge Protector");

        AddDuplicate(dupes, componentsBlock, "advanced_microprocessor", "Advanced Microprocessor");
        AddDuplicate(dupes, componentsBlock, "advanced_voltage_regulator", "Advanced Voltage Regulator");
        AddDuplicate(dupes, componentsBlock, "advanced_diode", "Advanced Diode");
        AddDuplicate(dupes, componentsBlock, "advanced_inductor", "Advanced Inductor");
        AddDuplicate(dupes, componentsBlock, "advanced_op_amp", "Advanced Op-Amp");

        AddDuplicate(dupes, componentsBlock, "broken_glitchy_wire", "Broken Glitchy Wire");
        AddDuplicate(dupes, componentsBlock, "broken_resistor", "Broken Resistor");
        AddDuplicate(dupes, componentsBlock, "broken_overloaded_capacitor", "Broken Overloaded Capacitor");
        AddDuplicate(dupes, componentsBlock, "broken_debugging_code", "Broken Debugging Code");

        AddDuplicate(dupes, componentsBlock, "quantum_resistor", "Quantum Resistor");
        AddDuplicate(dupes, componentsBlock, "quantum_overclocked_chip", "Quantum Overclocked Chip");
        AddDuplicate(dupes, componentsBlock, "quantum_crystal_oscillator", "Quantum Crystal Oscillator");
        AddDuplicate(dupes, componentsBlock, "quantum_rf_antenna", "Quantum RF Antenna");
        AddDuplicate(dupes, componentsBlock, "quantum_memory_chip", "Quantum Memory Chip");
        AddDuplicate(dupes, componentsBlock, "quantum_shielded_wire", "Quantum Shielded Wire");

        if (dupes.Length == 0)
        {
            CPH.SendMessage("@" + viewerName + " you don't have any duplicate components yet.", true, true);
            return true;
        }

        CPH.SendMessage("@" + viewerName + " Duplicates: " + dupes.ToString(), true, true);
        return true;
    }

    private void AddDuplicate(StringBuilder dupes, string componentsBlock, string componentId, string displayName)
    {
        int quantity = GetComponentQuantity(componentsBlock, componentId);

        if (quantity > 1)
        {
            if (dupes.Length > 0)
            {
                dupes.Append(" | ");
            }

            dupes.Append(displayName);
            dupes.Append(" x");
            dupes.Append(quantity);
        }
    }

    private string GetArgOrDefault(string argName, string defaultValue)
    {
        string value = "";

        if (CPH.TryGetArg(argName, out value))
        {
            if (!string.IsNullOrWhiteSpace(value))
            {
                return value;
            }
        }

        return defaultValue;
    }

    private int GetComponentQuantity(string text, string componentId)
    {
        string search = "\"" + componentId + "\"";
        int keyIndex = text.IndexOf(search, StringComparison.OrdinalIgnoreCase);

        if (keyIndex < 0)
        {
            return 0;
        }

        int colonIndex = text.IndexOf(":", keyIndex);

        if (colonIndex < 0)
        {
            return 0;
        }

        int numberStart = colonIndex + 1;

        while (numberStart < text.Length && Char.IsWhiteSpace(text[numberStart]))
        {
            numberStart++;
        }

        int numberEnd = numberStart;

        while (numberEnd < text.Length && Char.IsDigit(text[numberEnd]))
        {
            numberEnd++;
        }

        if (numberEnd <= numberStart)
        {
            return 0;
        }

        string numberText = text.Substring(numberStart, numberEnd - numberStart);

        int quantity = 0;

        if (Int32.TryParse(numberText, out quantity))
        {
            return quantity;
        }

        return 0;
    }

    private string ExtractJsonObjectForKey(string json, string key)
    {
        string search = "\"" + key + "\"";
        int keyIndex = json.IndexOf(search, StringComparison.OrdinalIgnoreCase);

        if (keyIndex < 0)
        {
            return "";
        }

        int colonIndex = json.IndexOf(":", keyIndex);

        if (colonIndex < 0)
        {
            return "";
        }

        int objectStart = json.IndexOf("{", colonIndex);

        if (objectStart < 0)
        {
            return "";
        }

        int depth = 0;

        for (int i = objectStart; i < json.Length; i++)
        {
            if (json[i] == '{')
            {
                depth++;
            }
            else if (json[i] == '}')
            {
                depth--;

                if (depth == 0)
                {
                    return json.Substring(objectStart, i - objectStart + 1);
                }
            }
        }

        return "";
    }
}
