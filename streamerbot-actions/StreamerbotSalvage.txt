using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;
using System.Threading;

public class CPHInline
{
    private const string SalvageUsageTemplate = "@{viewer} usage: !{salvageCommand} <{collectionSingular}> or !{salvageCommand} all";
    private const string NothingToSalvageTemplate = "@{viewer} you have no extra copies to salvage in {selection}.";
    private const string SalvageSuccessTemplate = "@{viewer} salvaged {count} extra {itemWord} for {earned} {currency}. Balance: {balance}.";

    public bool Execute()
    {
        string folderPath = @"C:\CircuitOS\Data";
        string componentsPath = Path.Combine(folderPath, "components.json");
        string inventoryPath = Path.Combine(folderPath, "inventory.json");
        string inventoryLockPath = Path.Combine(folderPath, "inventory.lock");
        string backupFolderPath = Path.Combine(folderPath, "backups");
        string viewerId = GetArgOrDefault("userId", "");
        string viewerName = GetViewerName();
        string requestedCollection = GetCommandInput();

        if (string.IsNullOrWhiteSpace(requestedCollection))
        {
            CPH.SendMessage(
                FormatMessage(SalvageUsageTemplate,
                    "viewer", viewerName,
                    "salvageCommand", "salvage",
                    "collectionSingular", "collection"),
                true,
                true
            );
            return true;
        }

        if (string.IsNullOrWhiteSpace(viewerId))
        {
            CPH.SendMessage("@" + viewerName + " I couldn't find your Twitch user ID.", true, true);
            return false;
        }

        if (!File.Exists(componentsPath) || !File.Exists(inventoryPath))
        {
            CPH.SendMessage("Salvage error: a Circuit Components data file is missing.", true, true);
            CPH.LogError("[Salvage] Missing components.json or inventory.json.");
            return false;
        }

        Dictionary<string, object> catalog;
        Dictionary<string, object> collections;

        try
        {
            catalog = RequireObject(SimpleJson.Parse(File.ReadAllText(componentsPath)), "catalog root");
            collections = GetObject(catalog, "collections");
        }
        catch (Exception ex)
        {
            CPH.SendMessage("Salvage error: components.json is invalid.", true, true);
            CPH.LogError("[Salvage] Catalog error: " + ex.Message);
            return false;
        }

        List<KeyValuePair<string, object>> selectedCollections;

        try
        {
            selectedCollections = ResolveCollections(collections, requestedCollection);
        }
        catch (Exception ex)
        {
            CPH.SendMessage("Salvage error: " + ex.Message, true, true);
            CPH.LogError("[Salvage] Selection error: " + ex.Message);
            return false;
        }

        if (selectedCollections.Count == 0)
        {
            CPH.SendMessage("@" + viewerName + " unknown collection.", true, true);
            return true;
        }

        FileStream inventoryLock = AcquireInventoryLock(inventoryLockPath);

        if (inventoryLock == null)
        {
            CPH.SendMessage("Salvage error: inventory is busy. Please try again.", true, true);
            return false;
        }

        using (inventoryLock)
        {
            Dictionary<string, object> inventory;

            try
            {
                inventory = RequireObject(SimpleJson.Parse(File.ReadAllText(inventoryPath)), "inventory root");
            }
            catch (Exception ex)
            {
                CPH.SendMessage("Salvage error: inventory.json is invalid.", true, true);
                CPH.LogError("[Salvage] Inventory error: " + ex.Message);
                return false;
            }

            Dictionary<string, object> viewer = TryGetObject(inventory, viewerId);
            Dictionary<string, object> viewerComponents = viewer == null
                ? null
                : TryGetObject(viewer, "components");

            if (viewer == null || viewerComponents == null)
            {
                CPH.SendMessage("@" + viewerName + " you don't have any components to salvage.", true, true);
                return true;
            }

            long consumedComponents = 0;
            long earnedScrap = 0;

            try
            {
                foreach (KeyValuePair<string, object> selected in selectedCollections)
                {
                    Dictionary<string, object> collection = RequireObject(selected.Value, selected.Key);
                    List<object> parts = GetArray(collection, "parts");
                    long salvageValue = GetSalvageValue(selected.Key, collection);

                    foreach (object partValue in parts)
                    {
                        Dictionary<string, object> part = RequireObject(partValue, selected.Key + " part");
                        string partId = GetString(part, "id");
                        object quantityValue;

                        if (string.IsNullOrWhiteSpace(partId) ||
                            !viewerComponents.TryGetValue(partId, out quantityValue))
                        {
                            continue;
                        }

                        long quantity = RequireInteger(quantityValue, partId);

                        if (quantity < 0)
                        {
                            throw new Exception("Negative component quantity: " + partId);
                        }

                        long extras = Math.Max(0L, quantity - 1L);

                        if (extras > 0)
                        {
                            viewerComponents[partId] = 1L;
                            consumedComponents = checked(consumedComponents + extras);
                            earnedScrap = checked(earnedScrap + checked(extras * salvageValue));
                        }
                    }
                }
            }
            catch (Exception ex)
            {
                CPH.SendMessage("Salvage error: inventory values could not be processed.", true, true);
                CPH.LogError("[Salvage] Processing error: " + ex.Message);
                return false;
            }

            if (consumedComponents == 0)
            {
                CPH.SendMessage(
                    FormatMessage(NothingToSalvageTemplate,
                        "viewer", viewerName,
                        "selection", GetSelectionLabel(selectedCollections, requestedCollection)),
                    true,
                    true
                );
                return true;
            }

            Dictionary<string, object> wallet = TryGetObject(viewer, "wallet");

            if (viewer.ContainsKey("wallet") && wallet == null)
            {
                CPH.SendMessage("Salvage error: wallet data is invalid.", true, true);
                return false;
            }

            if (wallet == null)
            {
                wallet = new Dictionary<string, object>();
                viewer["wallet"] = wallet;
            }

            object currentScrapValue;
            long currentScrap = wallet.TryGetValue("scrap", out currentScrapValue)
                ? RequireInteger(currentScrapValue, "wallet.scrap")
                : 0L;

            if (currentScrap < 0)
            {
                CPH.SendMessage("Salvage error: Scrap balance is invalid.", true, true);
                return false;
            }

            long newScrapBalance;

            try
            {
                newScrapBalance = checked(currentScrap + earnedScrap);
            }
            catch (OverflowException)
            {
                CPH.SendMessage("Salvage error: Scrap balance limit reached.", true, true);
                return false;
            }

            wallet["scrap"] = newScrapBalance;

            try
            {
                string backupPath = SaveInventorySafely(inventoryPath, backupFolderPath, inventory);
                CPH.LogInfo("[Salvage] Inventory backup created: " + backupPath);
            }
            catch (Exception ex)
            {
                CPH.SendMessage("Salvage error: inventory could not be saved.", true, true);
                CPH.LogError("[Salvage] Save error: " + ex.Message);
                return false;
            }

            CPH.SendMessage(
                FormatMessage(SalvageSuccessTemplate,
                    "viewer", viewerName,
                    "count", consumedComponents.ToString(),
                    "itemWord", consumedComponents == 1 ? "component" : "components",
                    "earned", earnedScrap.ToString(),
                    "currency", "Scrap",
                    "balance", newScrapBalance.ToString()),
                true,
                true
            );

            CPH.LogInfo(
                "[Salvage] " + viewerName + " consumed " + consumedComponents +
                " extras and earned " + earnedScrap + " Scrap."
            );

            return true;
        }
    }

    private List<KeyValuePair<string, object>> ResolveCollections(
        Dictionary<string, object> collections,
        string requestedCollection)
    {
        List<KeyValuePair<string, object>> selected = new List<KeyValuePair<string, object>>();
        string normalizedRequest = NormalizeCollectionName(requestedCollection);

        foreach (KeyValuePair<string, object> property in collections)
        {
            Dictionary<string, object> collection = RequireObject(property.Value, property.Key);

            if (normalizedRequest == "all" ||
                NormalizeCollectionName(property.Key) == normalizedRequest ||
                NormalizeCollectionName(GetString(collection, "displayName")) == normalizedRequest)
            {
                GetSalvageValue(property.Key, collection);
                selected.Add(property);

                if (normalizedRequest != "all")
                {
                    break;
                }
            }
        }

        return selected;
    }

    private long GetSalvageValue(string collectionKey, Dictionary<string, object> collection)
    {
        object configuredValue;

        if (collection.TryGetValue("salvageValue", out configuredValue))
        {
            long value = RequireInteger(configuredValue, collectionKey + ".salvageValue");
            if (value <= 0) throw new Exception("salvageValue must be positive: " + collectionKey);
            return value;
        }

        if (collectionKey == "basic") return 1;
        if (collectionKey == "power") return 2;
        if (collectionKey == "advanced") return 3;
        if (collectionKey == "broken") return 5;
        if (collectionKey == "quantum") return 10;

        throw new Exception("Collection is missing salvageValue: " + collectionKey);
    }

    private string GetSelectionLabel(
        List<KeyValuePair<string, object>> selectedCollections,
        string requestedCollection)
    {
        if (NormalizeCollectionName(requestedCollection) == "all") return "all collections";
        Dictionary<string, object> collection = RequireObject(selectedCollections[0].Value, "collection");
        string displayName = GetString(collection, "displayName");
        return string.IsNullOrWhiteSpace(displayName) ? selectedCollections[0].Key : displayName;
    }

    private Dictionary<string, object> GetObject(Dictionary<string, object> parent, string key)
    {
        Dictionary<string, object> value = TryGetObject(parent, key);
        if (value == null) throw new Exception("Missing JSON object: " + key);
        return value;
    }

    private Dictionary<string, object> TryGetObject(Dictionary<string, object> parent, string key)
    {
        object value;
        return parent != null && parent.TryGetValue(key, out value)
            ? value as Dictionary<string, object>
            : null;
    }

    private Dictionary<string, object> RequireObject(object value, string name)
    {
        Dictionary<string, object> result = value as Dictionary<string, object>;
        if (result == null) throw new Exception(name + " must be a JSON object.");
        return result;
    }

    private List<object> GetArray(Dictionary<string, object> parent, string key)
    {
        object value;
        List<object> result = parent.TryGetValue(key, out value) ? value as List<object> : null;
        if (result == null) throw new Exception(key + " must be a JSON array.");
        return result;
    }

    private string GetString(Dictionary<string, object> parent, string key)
    {
        object value;
        return parent.TryGetValue(key, out value) && value is string ? (string)value : "";
    }

    private long RequireInteger(object value, string name)
    {
        if (value is long) return (long)value;
        if (value is int) return (int)value;
        throw new Exception(name + " must be an integer.");
    }

    private string NormalizeCollectionName(string value)
    {
        StringBuilder normalized = new StringBuilder();

        foreach (char character in (value ?? "").ToLowerInvariant())
        {
            if (Char.IsLetterOrDigit(character)) normalized.Append(character);
        }

        string result = normalized.ToString();
        const string suffix = "collection";
        if (result.EndsWith(suffix, StringComparison.Ordinal))
            result = result.Substring(0, result.Length - suffix.Length);
        return result;
    }

    private string FormatMessage(string template, params string[] replacements)
    {
        string result = template ?? "";
        for (int i = 0; i + 1 < replacements.Length; i += 2)
        {
            result = result.Replace("{" + replacements[i] + "}", replacements[i + 1] ?? "");
        }
        return result;
    }

    private string GetCommandInput()
    {
        string input = GetArgOrDefault("rawInput", "");
        if (string.IsNullOrWhiteSpace(input)) input = GetArgOrDefault("input0", "");
        if (string.IsNullOrWhiteSpace(input)) input = GetArgOrDefault("argument0", "");
        input = input.Trim();
        if (input.StartsWith("!salvage", StringComparison.OrdinalIgnoreCase))
            input = input.Substring("!salvage".Length).Trim();
        return input;
    }

    private string GetViewerName()
    {
        string name = GetArgOrDefault("userName", "");
        if (string.IsNullOrWhiteSpace(name)) name = GetArgOrDefault("displayName", "");
        if (string.IsNullOrWhiteSpace(name)) name = GetArgOrDefault("user", "Viewer");
        return name;
    }

    private string GetArgOrDefault(string name, string defaultValue)
    {
        string value;
        return CPH.TryGetArg(name, out value) && !string.IsNullOrWhiteSpace(value)
            ? value
            : defaultValue;
    }

    private FileStream AcquireInventoryLock(string lockPath)
    {
        for (int attempt = 1; attempt <= 50; attempt++)
        {
            try
            {
                return new FileStream(lockPath, FileMode.OpenOrCreate, FileAccess.ReadWrite, FileShare.None);
            }
            catch (IOException)
            {
                if (attempt < 50) Thread.Sleep(100);
            }
        }

        return null;
    }

    private string SaveInventorySafely(
        string inventoryPath,
        string backupFolderPath,
        Dictionary<string, object> inventory)
    {
        string tempPath = Path.Combine(
            Path.GetDirectoryName(inventoryPath),
            "inventory." + Guid.NewGuid().ToString("N") + ".tmp"
        );

        try
        {
            string json = SimpleJson.Stringify(inventory);
            SimpleJson.Parse(json);
            File.WriteAllText(tempPath, json, new UTF8Encoding(false));
            SimpleJson.Parse(File.ReadAllText(tempPath));
            Directory.CreateDirectory(backupFolderPath);

            string backupPath = Path.Combine(
                backupFolderPath,
                "inventory_" + DateTime.Now.ToString("yyyyMMdd_HHmmss_fff") + ".json"
            );

            File.Replace(tempPath, inventoryPath, backupPath);
            return backupPath;
        }
        finally
        {
            if (File.Exists(tempPath))
            {
                try { File.Delete(tempPath); } catch { }
            }
        }
    }

    private class SimpleJson
    {
        private readonly string text;
        private int index;

        private SimpleJson(string json)
        {
            text = json ?? "";
        }

        public static object Parse(string json)
        {
            SimpleJson parser = new SimpleJson(json);
            object value = parser.ReadValue();
            parser.SkipWhitespace();
            if (parser.index != parser.text.Length) throw new Exception("Unexpected JSON content.");
            return value;
        }

        public static string Stringify(object value)
        {
            StringBuilder output = new StringBuilder();
            WriteValue(output, value, 0);
            output.AppendLine();
            return output.ToString();
        }

        private object ReadValue()
        {
            SkipWhitespace();
            if (index >= text.Length) throw new Exception("Unexpected end of JSON.");
            char character = text[index];
            if (character == '{') return ReadObject();
            if (character == '[') return ReadArray();
            if (character == '"') return ReadString();
            if (character == '-' || Char.IsDigit(character)) return ReadNumber();
            if (Match("true")) return true;
            if (Match("false")) return false;
            if (Match("null")) return null;
            throw new Exception("Invalid JSON value at position " + index + ".");
        }

        private Dictionary<string, object> ReadObject()
        {
            Dictionary<string, object> result = new Dictionary<string, object>();
            index++;
            SkipWhitespace();
            if (Take('}')) return result;

            while (true)
            {
                SkipWhitespace();
                if (index >= text.Length || text[index] != '"') throw new Exception("Expected object key.");
                string key = ReadString();
                if (result.ContainsKey(key)) throw new Exception("Duplicate JSON key: " + key);
                SkipWhitespace();
                Expect(':');
                result[key] = ReadValue();
                SkipWhitespace();
                if (Take('}')) return result;
                Expect(',');
            }
        }

        private List<object> ReadArray()
        {
            List<object> result = new List<object>();
            index++;
            SkipWhitespace();
            if (Take(']')) return result;

            while (true)
            {
                result.Add(ReadValue());
                SkipWhitespace();
                if (Take(']')) return result;
                Expect(',');
            }
        }

        private string ReadString()
        {
            Expect('"');
            StringBuilder result = new StringBuilder();

            while (index < text.Length)
            {
                char character = text[index++];
                if (character == '"') return result.ToString();
                if (character != '\\')
                {
                    if (character < 32) throw new Exception("Invalid control character in string.");
                    result.Append(character);
                    continue;
                }

                if (index >= text.Length) throw new Exception("Invalid string escape.");
                char escaped = text[index++];
                if (escaped == '"' || escaped == '\\' || escaped == '/') result.Append(escaped);
                else if (escaped == 'b') result.Append('\b');
                else if (escaped == 'f') result.Append('\f');
                else if (escaped == 'n') result.Append('\n');
                else if (escaped == 'r') result.Append('\r');
                else if (escaped == 't') result.Append('\t');
                else if (escaped == 'u')
                {
                    if (index + 4 > text.Length) throw new Exception("Invalid Unicode escape.");
                    int code;
                    if (!Int32.TryParse(text.Substring(index, 4), NumberStyles.HexNumber, CultureInfo.InvariantCulture, out code))
                        throw new Exception("Invalid Unicode escape.");
                    result.Append((char)code);
                    index += 4;
                }
                else throw new Exception("Invalid string escape.");
            }

            throw new Exception("Unterminated JSON string.");
        }

        private object ReadNumber()
        {
            int start = index;
            if (text[index] == '-') index++;
            if (index >= text.Length) throw new Exception("Invalid JSON number.");
            if (text[index] == '0') index++;
            else
            {
                if (!Char.IsDigit(text[index])) throw new Exception("Invalid JSON number.");
                while (index < text.Length && Char.IsDigit(text[index])) index++;
            }

            bool isFloatingPoint = false;
            if (index < text.Length && text[index] == '.')
            {
                isFloatingPoint = true;
                index++;
                if (index >= text.Length || !Char.IsDigit(text[index])) throw new Exception("Invalid JSON number.");
                while (index < text.Length && Char.IsDigit(text[index])) index++;
            }

            if (index < text.Length && (text[index] == 'e' || text[index] == 'E'))
            {
                isFloatingPoint = true;
                index++;
                if (index < text.Length && (text[index] == '+' || text[index] == '-')) index++;
                if (index >= text.Length || !Char.IsDigit(text[index])) throw new Exception("Invalid JSON exponent.");
                while (index < text.Length && Char.IsDigit(text[index])) index++;
            }

            string number = text.Substring(start, index - start);
            if (!isFloatingPoint)
            {
                long integer;
                if (Int64.TryParse(number, NumberStyles.Integer, CultureInfo.InvariantCulture, out integer)) return integer;
            }

            double floatingPoint;
            if (Double.TryParse(number, NumberStyles.Float, CultureInfo.InvariantCulture, out floatingPoint) &&
                !Double.IsNaN(floatingPoint) && !Double.IsInfinity(floatingPoint)) return floatingPoint;
            throw new Exception("Invalid JSON number.");
        }

        private bool Match(string value)
        {
            if (index + value.Length > text.Length ||
                String.CompareOrdinal(text, index, value, 0, value.Length) != 0) return false;
            index += value.Length;
            return true;
        }

        private void SkipWhitespace()
        {
            while (index < text.Length && Char.IsWhiteSpace(text[index])) index++;
        }

        private bool Take(char expected)
        {
            if (index < text.Length && text[index] == expected) { index++; return true; }
            return false;
        }

        private void Expect(char expected)
        {
            if (!Take(expected)) throw new Exception("Expected '" + expected + "' at position " + index + ".");
        }

        private static void WriteValue(StringBuilder output, object value, int depth)
        {
            if (value == null) { output.Append("null"); return; }
            Dictionary<string, object> objectValue = value as Dictionary<string, object>;
            if (objectValue != null) { WriteObject(output, objectValue, depth); return; }
            List<object> arrayValue = value as List<object>;
            if (arrayValue != null) { WriteArray(output, arrayValue, depth); return; }
            if (value is string) { WriteString(output, (string)value); return; }
            if (value is bool) { output.Append((bool)value ? "true" : "false"); return; }
            if (value is long) { output.Append(((long)value).ToString(CultureInfo.InvariantCulture)); return; }
            if (value is int) { output.Append(((int)value).ToString(CultureInfo.InvariantCulture)); return; }
            if (value is double)
            {
                double number = (double)value;
                if (Double.IsNaN(number) || Double.IsInfinity(number)) throw new Exception("Cannot write invalid number.");
                output.Append(number.ToString("R", CultureInfo.InvariantCulture));
                return;
            }
            throw new Exception("Unsupported JSON value type: " + value.GetType().FullName);
        }

        private static void WriteObject(StringBuilder output, Dictionary<string, object> value, int depth)
        {
            output.Append('{');
            bool first = true;
            foreach (KeyValuePair<string, object> property in value)
            {
                if (!first) output.Append(',');
                output.AppendLine();
                Indent(output, depth + 1);
                WriteString(output, property.Key);
                output.Append(": ");
                WriteValue(output, property.Value, depth + 1);
                first = false;
            }
            if (!first) { output.AppendLine(); Indent(output, depth); }
            output.Append('}');
        }

        private static void WriteArray(StringBuilder output, List<object> value, int depth)
        {
            output.Append('[');
            for (int i = 0; i < value.Count; i++)
            {
                if (i > 0) output.Append(',');
                output.AppendLine();
                Indent(output, depth + 1);
                WriteValue(output, value[i], depth + 1);
            }
            if (value.Count > 0) { output.AppendLine(); Indent(output, depth); }
            output.Append(']');
        }

        private static void WriteString(StringBuilder output, string value)
        {
            output.Append('"');
            foreach (char character in value)
            {
                if (character == '"') output.Append("\\\"");
                else if (character == '\\') output.Append("\\\\");
                else if (character == '\b') output.Append("\\b");
                else if (character == '\f') output.Append("\\f");
                else if (character == '\n') output.Append("\\n");
                else if (character == '\r') output.Append("\\r");
                else if (character == '\t') output.Append("\\t");
                else if (character < 32) output.Append("\\u" + ((int)character).ToString("x4"));
                else output.Append(character);
            }
            output.Append('"');
        }

        private static void Indent(StringBuilder output, int depth)
        {
            output.Append(' ', depth * 2);
        }
    }
}
