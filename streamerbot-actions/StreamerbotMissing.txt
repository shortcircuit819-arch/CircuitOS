using System;
using System.IO;
using System.Text;

public class CPHInline
{
    public bool Execute()
    {
        string folderPath = @"C:\Users\nicho\OneDrive\Documents\CircuitComponents";
        string inventoryPath = Path.Combine(folderPath, "inventory.json");

        if (!File.Exists(inventoryPath))
        {
            CPH.SendMessage("⚠️ Missing command error: inventory.json was not found.", true, true);
            return false;
        }

        string viewerId = GetArgOrDefault("userId", "");
        string viewerName = GetArgOrDefault("userName", "");

        if (string.IsNullOrWhiteSpace(viewerName))
        {
            viewerName = GetArgOrDefault("displayName", "");
        }

        if (string.IsNullOrWhiteSpace(viewerName))
        {
            viewerName = GetArgOrDefault("user", "Viewer");
        }

        if (string.IsNullOrWhiteSpace(viewerId))
        {
            CPH.SendMessage("@" + viewerName + " I couldn't find your Twitch user ID yet. Try redeeming a Circuit Component first.", true, true);
            return false;
        }

        string inventoryText = File.ReadAllText(inventoryPath);

        string viewerBlock = ExtractJsonObjectForKey(inventoryText, viewerId);

        if (string.IsNullOrWhiteSpace(viewerBlock))
        {
            CPH.SendMessage("@" + viewerName + " you don't have any Circuit Components yet. Redeem Circuit Component to start your collection.", true, true);
            return true;
        }

        string componentsBlock = ExtractJsonObjectForKey(viewerBlock, "components");

        if (string.IsNullOrWhiteSpace(componentsBlock))
        {
            CPH.SendMessage("@" + viewerName + " you don't have any Circuit Components yet. Redeem Circuit Component to start your collection.", true, true);
            return true;
        }

        string basic = GetMissingList(
            componentsBlock,
            new string[] { "basic_resistor", "basic_capacitor", "basic_wire", "basic_led", "basic_switch" },
            new string[] { "Resistor", "Capacitor", "Wire", "LED", "Switch" }
        );

        string power = GetMissingList(
            componentsBlock,
            new string[] { "power_battery", "power_capacitor", "power_fuse", "power_switch", "power_surge_protector" },
            new string[] { "Battery", "Capacitor", "Fuse", "Switch", "Surge Protector" }
        );

        string advanced = GetMissingList(
            componentsBlock,
            new string[] { "advanced_microprocessor", "advanced_voltage_regulator", "advanced_diode", "advanced_inductor", "advanced_op_amp" },
            new string[] { "Microprocessor", "Voltage Regulator", "Diode", "Inductor", "Op-Amp" }
        );

        string broken = GetMissingList(
            componentsBlock,
            new string[] { "broken_glitchy_wire", "broken_resistor", "broken_overloaded_capacitor", "broken_debugging_code" },
            new string[] { "Glitchy Wire", "Broken Resistor", "Overloaded Capacitor", "Debugging Code" }
        );

        string quantum = GetMissingList(
            componentsBlock,
            new string[] { "quantum_resistor", "quantum_overclocked_chip", "quantum_crystal_oscillator", "quantum_rf_antenna", "quantum_memory_chip", "quantum_shielded_wire" },
            new string[] { "Quantum Resistor", "Overclocked Chip", "Crystal Oscillator", "RF Antenna", "Memory Chip", "Shielded Wire" }
        );

        CPH.SendMessage("@" + viewerName + " Missing: Basic: " + basic + " | Power: " + power, true, true);
        CPH.SendMessage("@" + viewerName + " Missing: Advanced: " + advanced + " | Broken: " + broken + " | Quantum: " + quantum, true, true);

        return true;
    }

    private string GetArgOrDefault(string argName, string defaultValue)
    {
        string value = "";

        if (CPH.TryGetArg(argName, out value))
        {
            if (!string.IsNullOrWhiteSpace(value))
            {
                return value;
            }
        }

        return defaultValue;
    }

    private string GetMissingList(string componentsBlock, string[] componentIds, string[] componentNames)
    {
        StringBuilder missing = new StringBuilder();
        int missingCount = 0;

        for (int i = 0; i < componentIds.Length; i++)
        {
            int quantity = GetComponentQuantity(componentsBlock, componentIds[i]);

            if (quantity <= 0)
            {
                if (missingCount > 0)
                {
                    missing.Append(", ");
                }

                missing.Append(componentNames[i]);
                missingCount++;
            }
        }

        if (missingCount == 0)
        {
            return "COMPLETE";
        }

        return missing.ToString();
    }

    private int GetComponentQuantity(string text, string componentId)
    {
        string search = "\"" + componentId + "\"";
        int keyIndex = text.IndexOf(search, StringComparison.OrdinalIgnoreCase);

        if (keyIndex < 0)
        {
            return 0;
        }

        int colonIndex = text.IndexOf(":", keyIndex);

        if (colonIndex < 0)
        {
            return 0;
        }

        int numberStart = colonIndex + 1;

        while (numberStart < text.Length && Char.IsWhiteSpace(text[numberStart]))
        {
            numberStart++;
        }

        int numberEnd = numberStart;

        while (numberEnd < text.Length && Char.IsDigit(text[numberEnd]))
        {
            numberEnd++;
        }

        if (numberEnd <= numberStart)
        {
            return 0;
        }

        string numberText = text.Substring(numberStart, numberEnd - numberStart);

        int quantity = 0;

        if (Int32.TryParse(numberText, out quantity))
        {
            return quantity;
        }

        return 0;
    }

    private string ExtractJsonObjectForKey(string json, string key)
    {
        string search = "\"" + key + "\"";
        int keyIndex = json.IndexOf(search, StringComparison.OrdinalIgnoreCase);

        if (keyIndex < 0)
        {
            return "";
        }

        int colonIndex = json.IndexOf(":", keyIndex);

        if (colonIndex < 0)
        {
            return "";
        }

        int objectStart = json.IndexOf("{", colonIndex);

        if (objectStart < 0)
        {
            return "";
        }

        int depth = 0;

        for (int i = objectStart; i < json.Length; i++)
        {
            if (json[i] == '{')
            {
                depth++;
            }
            else if (json[i] == '}')
            {
                depth--;

                if (depth == 0)
                {
                    return json.Substring(objectStart, i - objectStart + 1);
                }
            }
        }

        return "";
    }
}
